import PostHogLayout from '../../../layouts/PostHogLayout.astro';
import { getCollection } from 'astro:content';
import '../../../styles/blog-cards.css';

export const allPosts = await getCollection('docs', ({ id, data }) => {
  // filter out index.mdx from the list of posts
  if (id.startsWith('blog/index.mdx')) {
    return false;
  }
  return id.startsWith('blog/');
});

export const blogEntries = [...allPosts].sort((a, b) => {
  const dateA = new Date(a.data.pubDate || '1970-01-01');
  const dateB = new Date(b.data.pubDate || '1970-01-01');
  return dateB - dateA;
});

<PostHogLayout />
<div class="blog-grid" style="margin-top: 1.5rem;">
  {blogEntries.map((post) => (
    <div class="blog-card">
      <a class="blog-card-link" href={`/${post.slug}`}>
        <h4>{post.data.title}</h4>
        {post.data.description && (<p>{post.data.description}</p>)}
        <div class="blog-meta">
          {post.data.pubDate && (
            <span class="date">{new Date(post.data.pubDate).toLocaleDateString('en-US', { year: 'numeric', month: 'short', day: 'numeric' })}</span>
          )}
          <span>•</span>
          <span class="author">{post.data.author || 'Code Pathfinder Team'}</span>
          <span style="margin-left:auto"></span>
          <span>Read →</span>
        </div>
      </a>
    </div>
  ))}
</div>
