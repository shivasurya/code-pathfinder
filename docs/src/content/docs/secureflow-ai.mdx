<script>
{`
  if (typeof window !== 'undefined') {
    // Global mode state
    window.currentEra = 'pre';
    
    // Switch between Reactive and Proactive modes
    window.switchEra = function(era) {
      window.currentEra = era;
      
      const btnPre = document.getElementById('btn-pre-llm');
      const btnPost = document.getElementById('btn-post-llm');
      const dinoFireball = document.getElementById('dino-fireball');
      const fallingFireball = document.getElementById('fireball-game');
      
      if (era === 'pre') {
        // Reactive Mode: Traditional SAST jumping over threats
        btnPre.style.background = '#535353';
        btnPre.style.color = '#fff';
        btnPost.style.background = 'transparent';
        btnPost.style.color = '#535353';
        
        // Change T-Rex label to SAST
        const dinoLabel = document.querySelector('.chrome-dino div[style*="margin-bottom"]');
        if (dinoLabel) dinoLabel.textContent = 'SAST';
        
        // Hide dinosaur shooting fireball
        if (dinoFireball) dinoFireball.style.display = 'none';
        
        // Show falling fireball (game over)
        if (fallingFireball) fallingFireball.style.display = 'block';
        
      } else {
        // Proactive Mode: AI-powered targeting and elimination
        btnPost.style.background = '#535353';
        btnPost.style.color = '#fff';
        btnPre.style.background = 'transparent';
        btnPre.style.color = '#535353';
        
        // Change T-Rex label to SecureFlow
        const dinoLabel = document.querySelector('.chrome-dino div[style*="margin-bottom"]');
        if (dinoLabel) dinoLabel.textContent = 'SecureFlow';
        
        // Show dinosaur shooting fireball
        if (dinoFireball) dinoFireball.style.display = 'block';
        
        // Hide falling fireball
        if (fallingFireball) fallingFireball.style.display = 'none';
        
        // Hide game over if showing
        const gameOver = document.getElementById('game-over');
        const gameDim = document.getElementById('game-dim');
        if (gameOver) gameOver.style.display = 'none';
        if (gameDim) gameDim.style.display = 'none';
        
        // Resume game if paused
        resumeGame();
        
        // Start making cacti disappear when hit
        startPostLLMMode();
      }
    };
    
    function startPostLLMMode() {
      // Make entire cactus container (including label) disappear when dinosaur shoots
      setTimeout(() => {
        const allDivs = document.querySelectorAll('div[style*="animation"]');
        allDivs.forEach(div => {
          const style = div.getAttribute('style');
          if (style && style.includes('obstacle-move')) {
            // Find the inner container with both label and image
            const innerContainer = div.querySelector('div[style*="text-align: center"]');
            if (innerContainer) {
              innerContainer.style.animation = 'cactus-disappear 0.5s ease-out forwards';
              setTimeout(() => {
                innerContainer.style.animation = '';
                innerContainer.style.opacity = '1';
                innerContainer.style.transform = 'scale(1)';
              }, 2000);
            }
          }
        });
        
        // Repeat for continuous effect
        if (window.currentEra === 'post') {
          setTimeout(startPostLLMMode, 2000);
        }
      }, 500);
    }
    
    function pauseGame() {
      // Pause T-Rex animation
      const dino = document.querySelector('.chrome-dino img');
      if (dino) {
        dino.style.animationPlayState = 'paused';
      }
      
      // Pause all cactus animations - find all animated divs
      const allDivs = document.querySelectorAll('div[style*="animation"]');
      allDivs.forEach(div => {
        const style = div.getAttribute('style');
        if (style && style.includes('obstacle-move')) {
          div.style.animationPlayState = 'paused';
        }
      });
    }
    
    function resumeGame() {
      // Resume T-Rex animation
      const dino = document.querySelector('.chrome-dino img');
      if (dino) {
        dino.style.animationPlayState = 'running';
      }
      
      // Resume all cactus animations
      const allDivs = document.querySelectorAll('div[style*="animation"]');
      allDivs.forEach(div => {
        const style = div.getAttribute('style');
        if (style && style.includes('obstacle-move')) {
          div.style.animationPlayState = 'running';
        }
      });
    }
    
    function restartGame() {
      const fireball = document.getElementById('fireball-game');
      const gameOver = document.getElementById('game-over');
      const gameDim = document.getElementById('game-dim');
      
      if (fireball && gameOver && gameDim) {
        // Hide game over message and dimming
        gameOver.style.display = 'none';
        gameDim.style.display = 'none';
        
        // Resume game animations
        resumeGame();
        
        // Reset fireball
        fireball.style.animation = 'none';
        fireball.style.opacity = '1';
        
        // Trigger reflow to restart animation
        void fireball.offsetWidth;
        
        // Restart animation
        fireball.style.animation = 'fireball-fall-single 10s ease-in forwards';
        
        // Schedule next game over (only in Reactive mode)
        setTimeout(() => {
          if (window.currentEra === 'pre') {
            // Pause game
            pauseGame();
            
            // Show game over
            gameDim.style.display = 'block';
            gameOver.style.display = 'block';
            
            setTimeout(restartGame, 2000);
          } else {
            // Proactive mode: no game over, continuous elimination
            setTimeout(restartGame, 0);
          }
        }, 10000);
      }
    }
    
    // Start the game cycle
    document.addEventListener('DOMContentLoaded', () => {
      setTimeout(() => {
        const gameOver = document.getElementById('game-over');
        const gameDim = document.getElementById('game-dim');
        if (gameOver && gameDim) {
          // Only show game over in Reactive mode
          if (window.currentEra === 'pre') {
            // Pause game
            pauseGame();
            
            // Show game over
            gameDim.style.display = 'block';
            gameOver.style.display = 'block';
            
            setTimeout(restartGame, 2000);
          }
        }
      }, 10000);
    });
  }
`}
</script>
